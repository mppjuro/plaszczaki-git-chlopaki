cmake_minimum_required(VERSION 3.10)
project(plaszczaki)

set(SOURCE_DIR "${CMAKE_SOURCE_DIR}/src")
set(CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/cmake_modules)

set(SDL2_INCLUDE_DIR "./lib/include")
set(SDL2_LIBRARIES "./lib/bin")
set(SDL2MAIN_LIBRARIES "./lib/lib")

include_directories(${SDL2_INCLUDE_DIR})
link_directories(${SDL2_LIBRARIES})
link_directories(${SDL2MAIN_LIBRARIES})

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra")

# Lista katalogów do stworzenia
set(DIRS
    ../data/input
    ../data/junk
    ../data/results
)

# Tworzenie katalogów
foreach(dir ${DIRS})
    file(MAKE_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/${dir})
endforeach()

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/..)
add_executable(plaszczaki WIN32 ${SOURCE_DIR}/main.cpp ${SOURCE_DIR}/functions.cpp)

target_link_libraries(${PROJECT_NAME} SDL2main SDL2 SDL2_ttf)
